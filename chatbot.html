                   "hola, como te va?",
                   "que hay?"]

                     "Nos vemos luego!",
                     "Fue un placer!",
                     "Cuidate",
                     "Bye"]

preguntas4=["como te llamas?"]
respuesta4=["LeyendItA","soy LeyendItA","mi nombre es LeyendItA"]
print("Hola soy LeyendItA, tu chatbot de confianza!"
      "si deseas terminar escribe: Salir")
while True:
    usuario=input("Tu:").lower().strip()

    if usuario in saludos:
        print("chatbot:",random.choice(respuesta_saludos))
    elif usuario in despedida:
        print("chatbot:",random.choice(respuesta_despedida))
        break
    elif usuario in preguntas1:
        print("chatbot:",random.choice(respuesta_preguntas1))
    elif usuario in preguntas2:
        print("chatbot:",random.choice(respuesta_preguntas2))
    elif usuario in preguntas3:
        print("chatbot:",random.choice(respuesta_preguntas3))
    elif usuario in preguntas4:
        print("chatbot:",random.choice(respuesta4))
